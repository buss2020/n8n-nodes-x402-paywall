# n8n X402 Paywall — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a working n8n community node that turns any workflow into a pay-per-call USDC-on-Arc endpoint via x402, plus 50+ real on-chain transactions as evidence and a 12-slide presentation — submission-ready on lablab.ai by 2026-04-25.

**Architecture:** n8n trigger node wraps incoming HTTP webhook, returns `402 Payment Required` when the `PAYMENT-SIGNATURE` header is absent, delegates verification and settlement to an external x402 facilitator, then triggers the downstream workflow with payment metadata. No private keys, no on-chain writes from the node itself.

**Tech Stack:** TypeScript, n8n community node interfaces (`n8n-workflow`), `@x402/*` TypeScript packages (version pinned in Task 2), `viem` for client-side signing in the burst script, `vitest` for unit tests, MIT license.

**Source spec:** `docs/superpowers/specs/2026-04-23-n8n-x402-paywall-design.md`

// nodes/X402Paywall/types.ts

/** Serialized shape of the `PAYMENT-REQUIRED` header after base64 decode. */
export interface PaymentRequiredBody {
  x402Version: 2;
  error: string;
  resource: {
    url: string;
    description: string;
    mimeType: string;
  };
  accepts: PaymentRequirementsEntry[];
}

export interface PaymentRequirementsEntry {
  scheme: "exact";
  network: string;              // "eip155:<chainId>"
  amount: string;               // base units as string (e.g. "5000" = $0.005 USDC)
  asset: string;                // USDC contract 0x...
  payTo: string;                // recipient 0x...
  maxTimeoutSeconds: number;
  extra: {
    name: "USDC";
    version: "2";
    facilitatorAddress?: string;
  };
}

/** Output emitted to downstream workflow on successful settlement. */
export interface SettlementOutput {
  txHash: string;
  payer: string;
  amount: string;
  amountUsd: string;
  network: string;
  asset: string;
  settledAt: string;            // ISO-8601
  facilitator: string;
}

/** Minimal shape of a facilitator /verify response (based on x402 spec). */
export interface FacilitatorVerifyResponse {
  valid: boolean;
  reason?: string;
  payer?: string;
  amount?: string;
}

/** Minimal shape of a facilitator /settle response. */
export interface FacilitatorSettleResponse {
  success: boolean;
  txHash?: string;
  payer?: string;
  amount?: string;
  settledAt?: string;
  reason?: string;
}

export class X402TimeoutError extends Error {
  constructor(stage: "verify" | "settle") {
    super(`X402 facilitator ${stage} timed out`);
    this.name = "X402TimeoutError";
  }
}

export class X402FacilitatorError extends Error {
  constructor(stage: "verify" | "settle", statusCode: number, body: string) {
    super(`X402 facilitator ${stage} returned ${statusCode}: ${body}`);
    this.name = "X402FacilitatorError";
  }
}

// nodes/X402Paywall/paymentRequirements.ts

import type { PaymentRequiredBody, PaymentRequirementsEntry } from "./types";
import type { NetworkConfig } from "./networks";

const USDC_DECIMALS = 6;

export function dollarsToBaseUnits(priceUsd: number): string {
  if (priceUsd <= 0 || !Number.isFinite(priceUsd)) {
    throw new Error(`Invalid price: ${priceUsd}. Must be positive finite number.`);
  }
  const base = Math.floor(priceUsd * 10 ** USDC_DECIMALS);
  if (base < 1) {
    throw new Error(`Price ${priceUsd} rounds below 1 base unit; increase price.`);
  }
  return String(base);
}

export interface BuildPaymentRequiredOptions {
  priceUsd: number;
  network: NetworkConfig;
  payTo: string;
  resourceUrl: string;
  description: string;
  mimeType: string;
  maxTimeoutSeconds: number;
  facilitatorAddress?: string;
}

export function buildPaymentRequiredBody(opts: BuildPaymentRequiredOptions): PaymentRequiredBody {
  const entry: PaymentRequirementsEntry = {
    scheme: "exact",
    network: opts.network.id,
    amount: dollarsToBaseUnits(opts.priceUsd),
    asset: opts.network.usdcContract,
    payTo: opts.payTo,
    maxTimeoutSeconds: opts.maxTimeoutSeconds,
    extra: {
      name: "USDC",
      version: "2",
      ...(opts.facilitatorAddress ? { facilitatorAddress: opts.facilitatorAddress } : {}),
    },
  };

  return {
    x402Version: 2,
    error: "Payment required",
    resource: {
      url: opts.resourceUrl,
      description: opts.description,
      mimeType: opts.mimeType,
    },
    accepts: [entry],
  };
}

export function encodePaymentRequiredHeader(body: PaymentRequiredBody): string {
  return Buffer.from(JSON.stringify(body), "utf-8").toString("base64");
}

export function decodePaymentSignatureHeader(header: string): unknown {
  return JSON.parse(Buffer.from(header, "base64").toString("utf-8"));
}
```

import { setupServer } from "msw/node";

import { FacilitatorClient } from "../nodes/X402Paywall/facilitatorClient";

const FAC_URL = "https://facilitator.test";
const server = setupServer();

beforeAll(() => server.listen({ onUnhandledRequest: "error" }));
afterEach(() => server.resetHandlers());
afterAll(() => server.close());

describe("FacilitatorClient.verify", () => {
  it("returns parsed valid=true on 200", async () => {
    server.use(
      http.post(`${FAC_URL}/verify`, () =>
        HttpResponse.json({ valid: true, payer: "0xabc", amount: "5000" }),
      ),
    );
    const client = new FacilitatorClient({ url: FAC_URL });
    const result = await client.verify({}, {});
    expect(result.valid).toBe(true);
    expect(result.payer).toBe("0xabc");

  it("returns valid=false with reason on rejection", async () => {
    server.use(

    const client = new FacilitatorClient({ url: FAC_URL });

    server.use(
      http.post(`${FAC_URL}/verify`, () =>
        HttpResponse.text("internal error", { status: 500 }),

    const client = new FacilitatorClient({ url: FAC_URL, maxRetries: 0 });
    await expect(client.verify({}, {})).rejects.toBeInstanceOf(X402FacilitatorError);

    server.use(
      http.post(`${FAC_URL}/verify`, async () => {
        await new Promise((r) => setTimeout(r, 500));
        return HttpResponse.json({ valid: true });

// nodes/X402Paywall/facilitatorClient.ts

import {
  FacilitatorVerifyResponse,
  FacilitatorSettleResponse,
  X402TimeoutError,
  X402FacilitatorError,
} from "./types";

export interface FacilitatorClientOptions {
  url: string;
  apiKey?: string;
  verifyTimeoutMs?: number;
  settleTimeoutMs?: number;
  maxRetries?: number;
}

export class FacilitatorClient {
  private readonly url: string;
  private readonly apiKey: string;
  private readonly verifyTimeoutMs: number;
  private readonly settleTimeoutMs: number;
  private readonly maxRetries: number;

  constructor(opts: FacilitatorClientOptions) {
    this.url = opts.url.replace(/\/+$/, "");
    this.apiKey = opts.apiKey ?? "";
    this.verifyTimeoutMs = opts.verifyTimeoutMs ?? 10_000;
    this.settleTimeoutMs = opts.settleTimeoutMs ?? 30_000;
    this.maxRetries = opts.maxRetries ?? 2;
  }

  async verify(paymentPayload: unknown, paymentRequirements: unknown): Promise<FacilitatorVerifyResponse> {
    return this.call<FacilitatorVerifyResponse>("/verify", { paymentPayload, paymentRequirements }, this.verifyTimeoutMs, "verify");
  }

  async settle(paymentPayload: unknown, paymentRequirements: unknown): Promise<FacilitatorSettleResponse> {
    return this.call<FacilitatorSettleResponse>("/settle", { paymentPayload, paymentRequirements }, this.settleTimeoutMs, "settle");
  }

  private async call<T>(path: string, body: unknown, timeoutMs: number, stage: "verify" | "settle"): Promise<T> {
    const endpoint = `${this.url}${path}`;
    let lastError: Error | undefined;

    for (let attempt = 0; attempt <= this.maxRetries; attempt++) {
      const controller = new AbortController();
      const timer = setTimeout(() => controller.abort(), timeoutMs);

      try {
        const res = await fetch(endpoint, {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            ...(this.apiKey ? { Authorization: `Bearer ${this.apiKey}` } : {}),
          },
          body: JSON.stringify(body),
          signal: controller.signal,
        });
        clearTimeout(timer);

        if (res.status >= 500) {
          const text = await res.text().catch(() => "");
          lastError = new X402FacilitatorError(stage, res.status, text);
          if (attempt < this.maxRetries) {
            await sleep(100 * 2 ** attempt);
            continue;
          }
          throw lastError;
        }

        if (!res.ok) {
          const text = await res.text().catch(() => "");
          throw new X402FacilitatorError(stage, res.status, text);
        }

        return (await res.json()) as T;
      } catch (err) {
        clearTimeout(timer);
        if ((err as Error).name === "AbortError") {
          throw new X402TimeoutError(stage);
        }
        if (err instanceof X402FacilitatorError) {
          throw err;
        }
        lastError = err as Error;
        if (attempt >= this.maxRetries) throw err;
        await sleep(100 * 2 ** attempt);
      }
    }
    throw lastError ?? new Error(`${stage} failed after ${this.maxRetries} retries`);
  }
}

function sleep(ms: number): Promise<void> {
  return new Promise((r) => setTimeout(r, ms));
}

// Usage: TARGET=http://localhost:5678/webhook/smoke COUNT=50 PRIVATE_KEY=0x... pnpm burst
//
// Fires N HTTP requests at an X402 Paywall endpoint. For each request:
//   1. Issues the initial unauthenticated GET
//   2. Decodes the 402 PAYMENT-REQUIRED body
//   3. Signs a payment payload (via @x402/evm ExactEvmScheme or viem fallback)
//   4. Retries with the PAYMENT-SIGNATURE header
//   5. Records the settled txHash from PAYMENT-RESPONSE

import { privateKeyToAccount } from "viem/accounts";
import { writeFile } from "node:fs/promises";

const TARGET = process.env.TARGET ?? "http://localhost:5678/webhook/smoke";
const COUNT = Number(process.env.COUNT ?? 50);
const PK = process.env.PRIVATE_KEY as `0x${string}` | undefined;
const EVIDENCE_FILE = process.env.EVIDENCE_FILE ?? "assets/burst-evidence.json";
const INTER_REQUEST_MS = Number(process.env.INTER_REQUEST_MS ?? 500);

if (!PK) {
  console.error("ERROR: PRIVATE_KEY env var required");
  process.exit(1);
}

interface Result {
  index: number;
  success: boolean;
  txHash?: string;
  durationMs: number;
  httpStatus: number;
  error?: string;
}

const account = privateKeyToAccount(PK);
console.log(`Burst: ${COUNT} requests to ${TARGET} from wallet ${account.address}`);

const results: Result[] = [];

for (let i = 0; i < COUNT; i++) {
  const start = Date.now();
  let httpStatus = 0;
  try {
    // Step 1: initial probe
    const probeRes = await fetch(TARGET);
    httpStatus = probeRes.status;
    if (probeRes.status !== 402) {
      throw new Error(`expected 402 on probe, got ${probeRes.status}`);
    }

    const reqHeader = probeRes.headers.get("payment-required");
    if (!reqHeader) throw new Error("no PAYMENT-REQUIRED header");
    const requirements = JSON.parse(Buffer.from(reqHeader, "base64").toString("utf-8"));
    const accept = requirements.accepts[0];

    // Step 2: sign payment payload
    // For @x402/evm ExactEvmScheme: await client.createPaymentPayload(requirements).
    // Fallback: the facilitator spec expects EIP-712 signature. Use provider integration.
    //
    // Simplified signing via @x402/evm (assumes package available):
    const { ExactEvmScheme } = await import("@x402/evm/exact/client");
    const scheme = new ExactEvmScheme(account);
    const payload = await scheme.createPaymentPayload(accept);

    // Step 3: retry with signature
    const sigHeader = Buffer.from(JSON.stringify(payload), "utf-8").toString("base64");
    const paidRes = await fetch(TARGET, {
      headers: { "PAYMENT-SIGNATURE": sigHeader },
    });
    httpStatus = paidRes.status;

    if (!paidRes.ok) {
      const body = await paidRes.text();
      throw new Error(`paid request ${paidRes.status}: ${body.slice(0, 200)}`);
    }

    const paymentRespHeader = paidRes.headers.get("payment-response");
    const txHash = paymentRespHeader
      ? JSON.parse(Buffer.from(paymentRespHeader, "base64").toString("utf-8")).txHash
      : undefined;

    const durationMs = Date.now() - start;
    results.push({ index: i + 1, success: true, txHash, durationMs, httpStatus });
    console.log(`[${i + 1}/${COUNT}] OK tx=${txHash} ${durationMs}ms`);
  } catch (err) {
    const durationMs = Date.now() - start;
    const msg = (err as Error).message;
    results.push({ index: i + 1, success: false, durationMs, httpStatus, error: msg });
    console.log(`[${i + 1}/${COUNT}] FAIL ${msg} (${durationMs}ms)`);
  }

  if (i < COUNT - 1 && INTER_REQUEST_MS > 0) {
    await new Promise((r) => setTimeout(r, INTER_REQUEST_MS));
  }
}

const ok = results.filter((r) => r.success).length;
const avgMs = Math.round(results.reduce((a, r) => a + r.durationMs, 0) / results.length);
console.log(`\n=== SUMMARY ===`);
console.log(`${ok}/${COUNT} successful. Avg ${avgMs}ms.`);

await writeFile(
  EVIDENCE_FILE,
  JSON.stringify(
    {
      ranAt: new Date().toISOString(),
      target: TARGET,
      wallet: account.address,
      count: COUNT,
      successful: ok,
      avgMs,
      results,
    },
    null,
    2,
  ),
);
console.log(`Evidence written to ${EVIDENCE_FILE}`);
```

## 🧩 How it works

1. Client requests your webhook URL

3. Client signs a payment payload, retries with `PAYMENT-SIGNATURE`
4. Node calls facilitator `/verify` → if valid, calls `/settle` → onchain confirmation
5. Workflow triggers with payment metadata (txHash, payer, amount)
6. Client receives `200` + workflow output + `PAYMENT-RESPONSE` header
